### index.js
---

```jsx
import Root from './routes/root';
import Home from './routes/home';
import Products from './routes/products';
import Cart from './routes/cart';
import Product from './routes/product';
  
const router = createBrowserRouter([
	{
		path: '/',
		element: <Root />,
		children: [
			{ index: true, element: <Home /> },
			{
				path: '/products',
				element: <Products />,
			},
			{
				path: '/cart',
				element: <Cart />,
			},
			{
				path: '/products/:id',
				element: <Product />,
			},
		],
	},
]);

const rootElement = document.getElementById('root');
const root = createRoot(rootElement);
root.render(
	<StrictMode>
		<RouterProvider router={router} />
	</StrictMode>
);
```
### products.hook.js
---

```jsx
import React, { useState, useEffect } from 'react';
  
export const useProducts = () => {
const [loading, loading_set] = useState(false);
const [products, products_set] = useState([]);
  
useEffect(() => {
	(async () => {
		loading_set(true);
		const response = await fetch('https://dummyjson.com/products');
		const json = await response.json();
		loading_set(false);
		products_set(json.products);
	})();
}, []);
  
return { loading, products };
};
  
export const useProduct = (id) => {
const [loading, loading_set] = useState(false);
const [product, product_set] = useState(null);
  
useEffect(() => {
	(async () => {
		loading_set(true);
		const response = await fetch('https://dummyjson.com/products/' + id);
		const json = await response.json();
		loading_set(false);
		product_set(json);
	})();
}, [id]);
  
return { loading, product };
};
```

```jsx
const Products = () => {
const { loading, products } = useProducts();
	return (
		<div>
			<hgroup>
				<h2>Products</h2>
				<p>Products page</p>
			</hgroup>
			<article aria-busy={loading}>
				<ul>
					{products?.map(({ id, title }) => (
					<li key={id}>
					<NavLink to={`/products/${id}`}>{title}</NavLink>
					</li>
					))}
				</ul>
			</article>
		</div>
	);
};
```

### product.js
---

```jsx
import React from 'react';
import { useParams } from 'react-router-dom';
import { useProduct } from '../hooks/products.hook';

const Product = () => {
	const { id } = useParams();
	const { loading, product } = useProduct(id);
	const { title, description, thumbnail } = product ?? {};
	return (
	<div>
		<hgroup>
			<h3>Prduct</h3>
			<p>Produc page</p>
		</hgroup>
		<article aria-busy={loading}>
			<header>
				<strong>{title}</strong>
			</header>
			<img src={thumbnail} style={{ width: '100%' }} />
			<footer>{description}</footer>
		</article>
	</div>
	);
};
  
export default Product;
```
